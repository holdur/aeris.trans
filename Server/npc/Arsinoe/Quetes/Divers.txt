	disablenpc "Lophophora#ep7";

	monster "ars_fild12",299,125,""+getmonsterinfo(3835,0)+"",3835,1,"EP7_Plante::OnMonsterKill"; 
	set $@monster_ep7plante,1;
	end;

OnMonsterKill:

	set $@monster_ep7plante, $@monster_ep7plante-1;
	if($@monster_ep7plante==0) {
		donpcevent "EP7_Plante::OnEnd";
	}
	end;

	if($ep7_LangChoice & 1) {mapannounce "ars_fild12","Vous pouvez désormais cueillir la Lophophora",0,0x66CC33;}
	if($ep7_LangChoice & 2) {mapannounce "ars_fild12","Now, You can pick the Lophophora",0,0x66CC33;}
	if($ep7_LangChoice & 4) {mapannounce "ars_fild12","Ahora puede tomar la Lophophora",0,0x66CC33;}
	set $@EP7_plante,2;
	enablenpc "Lophophora#ep7";
	sleep2 180000;
	donpcevent "EP7_Plante::OnInit";
	end;
}

	set EP7_FETE,2;
	changequest 70210,70211;
	end;

	mes "Alors cette viande de Curupira ? Elle arrive ? Tu n'as rapporté que " + EP7_FETE + " curupira, il t'en faut 50 !";
	close;

	mes "- 50 Bananes";
	mes "- 40 Pommes";
	mes "- 52 Oranges";
	mes "- 15 Mastela Fruit";
	mes "- 20 Prickly Fruit";
	next;
	mes "[" + @name$ + "]"; 
	mes "Je sais, je sais, ça en fait des courses à faire mais je te laisse imaginer quel grandiose festin nous allons faire !";	
	close2;
	set EP7_FETE,51;
	changequest 70211,70212;
	end;

	mes "[" + @name$ + "]"; 
	mes "Alors, voyons voir ce que tu m'as rapporté :";
	next;

function	script	arsi_pecheR	{

	//Reset
	set $@timeout,0;
	//Geste
	switch( rand(3) ) {
		case 0:set @geste$,"Haut";break;
		case 1:set @geste$,"Bas";break;
		case 2:set @geste$,"Gauche";break;
		case 3:set @geste$,"Droite";break;}

	//Temps de Pêche
	set @rand, rand(2000,5000);
	set $@ep7_time, rand(2000,4500);

	//Texte
	mes "[" + @name$ + "]";
	mes "Le Poisson approche ...";
	sleep2 @rand;
	mes "Vous devez tirer votre ligne : "+@geste$+" !";

	//Time out
	donpcevent "EP7_FishTime::OnTimeOut";

	//Message
	soundeffect "crab_stand.wav",0;
	set @Tgeste$,@geste$;
	input @Tgeste$;
	if (@Tgeste$ == @geste$) goto L_Bon;
	if (@Tgeste$ != @geste$) goto L_Bad;
	end;


L_Bon:
	//Réussite
	if ($@timeout == 1) goto L_Bad2;
	mes " ";
	mes "Vous venez de pêcher un "+@poisson$+" !";
	close2;
	delitem 29073,1;
	getitem @poissonid,1;
	end;

L_Bad2:

	mes " ";
	mes "Trop tard, le poisson est parti !"; close2; delitem 29073,1; end;
	close;

L_Bad:

	mes " ";
	mes "Vous n'avez rien au bout de votre ligne."; close2; delitem 29073,1; end;


}
