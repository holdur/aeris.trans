	monster "coppa",160,339,""+getmonsterinfo(2068,0)+"",2068,1,"EP7_Boitata::OnMonsterKill"; //Boitata

	end;

	set $@monster_ep7boitata, $@monster_ep7boitata-1;
	if($@monster_ep7boitata==0) {
		donpcevent "EP7_Boitata::OnEnd";}
	end;

		if($ep7_LangChoice & 1) {mapannounce "coppa","Servante Kloiacha : Retournez à la ville pour voir le chef ! Je te remercie pour ton aide !",0,0x66CC33;}
		if($ep7_LangChoice & 2) {mapannounce "coppa","Servante Kloiacha : Go back to town to see the chief, I thank you for your help!",0,0x66CC33;}
		if($ep7_LangChoice & 4) {mapannounce "coppa","Servante Kloiacha : Vuelve a ver al Alcande, el te lo agradecera!",0,0x66CC33;}
		set $@EP7_Boitata,2;
		sleep2 120000;
		donpcevent "EP7_Boitata::OnInit";
		end;
