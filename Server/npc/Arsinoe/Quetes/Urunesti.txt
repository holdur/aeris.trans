	// warp joueurs
	donpcevent "Infirmier Urluc#ep7::OnEnterBG";
	if( $EP7Annonce == 0 && $ep7_LangChoice & 1) {mapannounce "ars_fild42","Instance -- Le Sang du Shogunka vient de débuter !",1,0xFF6600;}
